                            <div><label className="text-xs font-bold text-gray-700 block mb-1">Startdatum</label><input type="date" className="w-full border rounded-lg p-3 bg-gray-50 focus:bg-white focus:ring-2 focus:ring-emerald-500 outline-none transition-all" value={formData.date} onChange={e=>setFormData({...formData, date: e.target.value})} /></div>
                            
                            <div className="pt-6 flex gap-4">
                                <button type="button" onClick={onCancel} className="flex-1 py-3 text-gray-500 hover:bg-gray-100 rounded-lg font-medium">Abbrechen</button>
                                <button type="submit" className="flex-1 py-3 bg-emerald-600 hover:bg-emerald-700 text-white rounded-lg font-bold shadow-lg shadow-emerald-200">Projekt erstellen</button>
                            </div>
                        </form>
                    </div>
                </div>
            );
        };

        // --- MAIN APP (PROJECT MANAGER) ---
        const App = () => {
          const [view, setView] = useState('list'); // 'list', 'onboarding', 'workspace'
          const [projects, setProjects] = useState(() => {
              const saved = localStorage.getItem('brandBuilder_projects');
              return saved ? JSON.parse(saved) : {};
          });
          const [activeProjectId, setActiveProjectId] = useState(null);
          
          // Workspace State (lifts up from previous App component)
          const [currentTool, setCurrentTool] = useState('dashboard');

          useEffect(() => {
              localStorage.setItem('brandBuilder_projects', JSON.stringify(projects));
          }, [projects]);

          const handleCreateProject = (metaData) => {
              const newProject = {
                  ...metaData,
                  progress: { business: 0, dna: 0, positioning: 0, structure: 0 },
                  // Empty data containers for modules
                  dna: { pool: [{ id: 'w1', text: 'Innovation', layer: 0 }, { id: 'w2', text: 'Nachhaltigkeit', layer: 1 }, { id: 'w3', text: 'Qualität', layer: 2 }], placedItems: [] },
                  positioning: { labels: { top: 'Premium', bottom: 'Budget', left: 'Tradition', right: 'Innovation' }, logos: [], arrows: [] },
                  structure: { root: { id: 'root', name: metaData.name, type: 'master', isOpen: true, children: [] } },
                  business: {},
                  target: {},
                  message: {}
              };
              setProjects(prev => ({ ...prev, [newProject.id]: newProject }));
              setActiveProjectId(newProject.id);
              setView('workspace');
          };

          const handleDeleteProject = (id) => {
              if (confirm("Projekt wirklich löschen?")) {
                  const newProjects = { ...projects };
                  delete newProjects[id];
                  setProjects(newProjects);
              }
          };

          const handleUpdateActiveProject = (updatedData) => {
              setProjects(prev => ({
                  ...prev,
                  [activeProjectId]: { ...prev[activeProjectId], ...updatedData }
              }));
          };

          // --- RENDER ---
          if (view === 'list') {
              return <ProjectListScreen projects={projects} onNew={() => setView('onboarding')} onSelect={(id) => { setActiveProjectId(id); setView('workspace'); setCurrentTool('dashboard'); }} onDelete={handleDeleteProject} />;
          }

          if (view === 'onboarding') {
              return <OnboardingScreen onFinish={handleCreateProject} onCancel={() => setView('list')} />;
          }

          if (view === 'workspace' && activeProjectId && projects[activeProjectId]) {
              const activeProject = projects[activeProjectId];
              
              // Workspace Router
              return (
                <div className="h-screen w-full bg-gray-50 font-sans text-gray-800">
                  {currentTool === 'dashboard' && <Dashboard onNavigate={setCurrentTool} projectData={activeProject} />}
                  {currentTool === 'business' && <BusinessModelTool onBack={() => setCurrentTool('dashboard')} globalData={activeProject} onUpdate={handleUpdateActiveProject} />}
                  {currentTool === 'positioning' && <PositioningTool onBack={() => setCurrentTool('dashboard')} globalData={activeProject} onUpdate={handleUpdateActiveProject} />}
                  {currentTool === 'dna' && <BrandDNATool onBack={() => setCurrentTool('dashboard')} globalData={activeProject} onUpdate={handleUpdateActiveProject} />}
                  {currentTool === 'structure' && <BrandStructureTool onBack={() => setCurrentTool('dashboard')} globalData={activeProject} onUpdate={handleUpdateActiveProject} />}
                  {currentTool === 'target' && <TargetAudienceTool onBack={() => setCurrentTool('dashboard')} globalData={activeProject} onUpdate={handleUpdateActiveProject} />}
                  {currentTool === 'naming' && <NamingTool onBack={() => setCurrentTool('dashboard')} />}
                  {currentTool === 'message' && <MessagingTool onBack={() => setCurrentTool('dashboard')} globalData={activeProject} onUpdate={handleUpdateActiveProject} />}
                  {currentTool === 'brandbook' && <BrandBook onBack={() => setCurrentTool('dashboard')} data={activeProject} />}
                  
                  {/* Global Back to Projects Link (optional, maybe in header) */}
                  <div className="fixed bottom-4 left-4 z-50 no-print">
                      <button onClick={() => setView('list')} className="bg-slate-800 text-white p-2 rounded-full shadow-lg hover:bg-slate-900" title="Zurück zur Projektübersicht"><Icons.FolderOpen size={16}/></button>
                  </div>
                </div>
              );
          }

          return <div>Error: Project state invalid. <button onClick={() => setView('list')}>Reset</button></div>;
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
